    /* ═══ SMART INJECT MODAL ═══ */
    const AddModal = ({ onClose, onSave }) => {
        const [inputVal, setInputVal] = useState(''); 
        const [fetching, setFetching] = useState(false);
        
        const handleSave = async () => {
            const val = inputVal.trim();
            if (!val) return;
            setFetching(true);
            
            // Smart auto-detect logic
            const isEmbed = /<iframe|<blockquote|<script|<embed|<object/i.test(val);
            const isUrl = (/^https?:\/\//i.test(val) || /^www\./i.test(val) || /^[^\s]+(?:\.com|\.net|\.org|\.io|\.xyz|\.co|\.me|\.m3u8|\.mp4|\.jpg|\.png|\.gif|\.webp|\.avif)(?:\/|$)/i.test(val)) && !/\s/.test(val);

            if (isEmbed) {
                await onSave({
                    url: '', content: val, type: 'embed', site: 'Embedded', collection: 'RAW', thumb: null, title: 'Embedded Content',
                    duration: null, isPortrait: false, tags: ['embed'], artist: '', character: '', source: '', description: ''
                });
            } else if (isUrl) {
                const safeUrl = val.startsWith('http') ? val : `https://${val}`;
                const type = getMediaType(safeUrl); let site = getDomain(safeUrl);
                let finalThumb = null; let finalDuration = null; let finalIsPortrait = false;
                if (['video','hls'].includes(type)) {
                    const vMeta = await generateVideoMeta(safeUrl, type);
                    finalThumb = vMeta.thumb; finalDuration = vMeta.duration; finalIsPortrait = vMeta.isPortrait;
                }
                await onSave({ 
                    url: safeUrl, type, site, collection: 'RAW', thumb: finalThumb, title: 'New Item', duration: finalDuration, 
                    isPortrait: finalIsPortrait, tags: [type], artist: '', character: '', source: '', description: ''
                });
            } else {
                const titleMatch = val.split('\n')[0].substring(0, 40);
                const title = titleMatch + (val.length > 40 ? '...' : '');
                await onSave({
                    url: '', content: val, type: 'note', site: 'Note', collection: 'Notes', thumb: null, title: title || 'Untitled Note',
                    duration: null, isPortrait: false, tags: ['note'], artist: '', character: '', source: '', description: val.substring(0, 100)
                });
            }
            
            setFetching(false); onClose();
        };

        return (
            <div className="fixed inset-0 z-[500] bg-[var(--overlay)] backdrop-blur-md flex items-end sm:items-center justify-center sm:p-4" onClick={e => e.target===e.currentTarget && onClose()}>
                <div className="w-full max-w-md glass-heavy rounded-t-[32px] sm:rounded-[32px] p-8 slide-up border border-[var(--border-color)] shadow-2xl">
                    <div className="flex justify-between items-center mb-6">
                        <h2 className="syne text-xl font-bold uppercase tracking-widest text-[var(--text-main)]">Inject Asset</h2>
                        <button onClick={onClose} className="w-10 h-10 bg-[var(--bg-input)] rounded-full flex items-center justify-center hover:bg-[var(--border-color)] spring-anim active-bounce"><X className="w-5 h-5"/></button>
                    </div>
                    
                    <textarea 
                        className="w-full p-5 rounded-[24px] mb-6 bg-[var(--bg-input)] border border-[var(--border-color)] outline-none focus:border-[var(--text-main)] spring-anim text-sm font-bold min-h-[160px] resize-none shadow-inner" 
                        placeholder="Write a note, paste a link (m3u8, web), or drop embed HTML code..." 
                        value={inputVal} 
                        onChange={e=>setInputVal(e.target.value)} 
                    />
                    <button onClick={handleSave} disabled={fetching} className="btn-action btn-primary active-bounce text-sm shadow-xl shadow-accent/20 touch-ripple">{fetching ? <Loader className="w-5 h-5 animate-spin"/> : 'Upload Node'}</button>

                                            layoutMode === 'compact' ? 'aspect-square rounded-[20px] hover-bounce active-bounce' : 
                                            isBento ? (bentoWide ? 'col-span-2 aspect-video rounded-[32px] hover-bounce active-bounce' : 'col-span-1 aspect-square rounded-[32px] hover-bounce active-bounce') :
                                            isFeed ? 'flex flex-col rounded-[32px] mb-4 bg-[var(--bg-modal)] shadow-lg active-bounce' : ''
                                        }`}>
                                            <div className={`relative bg-black/40 ${isList ? 'w-28 h-full shrink-0' : isFeed ? 'w-full aspect-[4/5] sm:aspect-[3/4]' : 'w-full h-full'}`}>
                                                {item.thumb || (item.type === 'image' ? item.url : null) ? (
                                                    <img src={item.thumb || item.url} className={`w-full h-full ${isFeed ? 'object-cover' : 'object-cover'}`} loading="lazy" />
                                                ) : item.type === 'embed' ? (
                                                    <div className="w-full h-full relative bg-black overflow-hidden">
                                                        <iframe 
                                                            srcDoc={`<!DOCTYPE html><html><head><meta charset="utf-8"><meta name="viewport" content="width=device-width, initial-scale=1"><style>body{margin:0;padding:12px;color:#f0f0f0;font-family:'Space Mono',monospace;display:flex;flex-direction:column;justify-content:center;align-items:center;min-height:90vh;background:#000;text-align:center;overflow:hidden;}h1,h2,h3{font-family:'Syne',sans-serif;margin:0 0 8px 0;font-size:2rem;color:white;}p{margin:0;color:#888899;font-weight:bold;text-transform:uppercase;letter-spacing:1px;font-size:12px;}</style></head><body><div id="weather"></div><div id="app"></div><div id="root"></div>${item.content}</body></html>`}
                                                            className="w-full h-full border-none pointer-events-none" sandbox="allow-scripts allow-same-origin" scrolling="no"
                                                        />
                                                    </div>
                                                ) : (
                                                    <div className="w-full h-full flex flex-col items-center justify-center p-2 opacity-60">
                                                        {item.type === 'note' ? <FileText className="w-8 h-8 text-accent"/> : <LinkIcon className="w-8 h-8 text-[var(--text-muted)]"/>}
                                                        {(isBento && !bentoWide) || isFeed && <span className="text-[10px] font-bold mt-2 uppercase tracking-widest text-[var(--text-muted)] truncate max-w-full px-2">{getDomain(item.site || item.url)}</span>}
                                                    </div>
                                                )}
                                                <div className="absolute top-3 right-3 flex gap-1 z-10">
                                                    {item.pinned && <div className="w-8 h-8 rounded-full bg-accent/90 backdrop-blur flex items-center justify-center shadow-lg"><Bookmark className="w-4 h-4 text-white fill-white"/></div>}
                                                    {['video','hls','youtube'].includes(item.type) && <div className="w-8 h-8 rounded-full bg-black/80 backdrop-blur flex items-center justify-center shadow-lg"><Play className="w-4 h-4 text-white fill-white ml-0.5"/></div>}
                                                    {item.type === 'note' && <div className="w-8 h-8 rounded-full bg-[var(--text-main)] backdrop-blur flex items-center justify-center shadow-lg"><FileText className="w-4 h-4 text-[var(--bg-base)]"/></div>}
                                                </div>
                                                {item.duration && (
                                                    <div className="absolute bottom-3 right-3 z-10">
                                                        <span className="bg-black/80 backdrop-blur-md px-2 py-1 rounded-md text-[10px] text-white font-bold tracking-widest shadow-lg">{formatDuration(item.duration)}</span>
                                                    </div>
                                                )}
                                            </div>

                                            {isList ? (
                                                <div className="p-4 flex flex-col justify-center flex-1 min-w-0">
                                                    <h4 className="text-sm font-bold truncate text-[var(--text-main)] mb-1">{item.title || 'Untitled Node'}</h4>
                                                    {item.type !== 'note' && item.type !== 'embed' && <p className="text-[10px] text-[var(--text-muted)] uppercase tracking-widest truncate">{getDomain(item.site || item.url)}</p>}
                                                </div>
                                            ) : isFeed ? (
                                                <div className="p-6 flex flex-col gap-4">
                                                    <div className="flex items-center gap-4 border-b border-[var(--border-color)] pb-4">
                                                        <Heart className={`w-6 h-6 ${item.liked ? 'fill-accent text-accent' : 'text-[var(--text-muted)]'}`} />
                                                        <Bookmark className={`w-6 h-6 ${item.pinned ? 'fill-accent text-accent' : 'text-[var(--text-muted)]'}`} />
                                                        <MessageSquare className="w-6 h-6 text-[var(--text-muted)]" />
                                                        <div className="flex-1"></div>
                                                        {item.type !== 'note' && item.type !== 'embed' && <span className="text-[10px] text-[var(--text-muted)] uppercase tracking-widest font-bold">{getDomain(item.site || item.url)}</span>}
                                                    </div>
                                                    <div>
                                                        <h4 className="text-xl font-bold text-[var(--text-main)] leading-tight">{item.title || 'Untitled Node'}</h4>
                                                        <span className="text-[10px] text-[var(--text-muted)] uppercase tracking-widest block mt-2">{new Date(item.savedAt||0).toLocaleString()}</span>
                                                    </div>
                                                </div>
                                            ) : (
                                                <div className="absolute inset-x-0 bottom-0 p-4 pt-16 bg-gradient-to-t from-black/95 via-black/60 to-transparent flex flex-col justify-end pointer-events-none">
                                                    <h4 className="text-xs md:text-sm font-bold text-white truncate drop-shadow-md">{item.title || 'Untitled Node'}</h4>
                                                    {isBento && bentoWide && item.type !== 'note' && item.type !== 'embed' && <p className="text-[10px] text-white/70 uppercase tracking-widest mt-1 font-bold">{item.type} // {getDomain(item.site || item.url)}</p>}
                                                </div>
                                            )}
                                        </div>
                                    );
                                })}
                            </div>
                        )}
                        {filteredItems.length === 0 && <div className="text-center mt-32 opacity-40 font-bold uppercase tracking-widest text-sm">Nothing Found</div>}
                    </main>
                )}

                {/* === SEARCH TAB === */}
                {currentTab === 'search' && (
                    <div className="max-w-4xl mx-auto p-4 pt-8 animate-in pb-32">
                        <div className="bg-[var(--bg-input)] rounded-full flex items-center px-6 h-16 border border-[var(--border-color)] focus-within:border-[var(--text-main)] transition-colors mb-8 shadow-sm">
                            <Search className="w-6 h-6 text-[var(--text-main)] mr-3" />
                            <input autoFocus className="w-full bg-transparent outline-none font-bold text-lg text-[var(--text-main)]" placeholder="Search ideas, pins, tags..." value={search} onChange={e=>setSearch(e.target.value)} />
                            {search && <button onClick={()=>setSearch('')} className="active-bounce"><X className="w-6 h-6 text-[var(--text-muted)]"/></button>}
                        </div>

                        {!search ? (
                            <div>
                                <h3 className="text-sm font-bold mb-4 px-2 text-[var(--text-main)]">Ideas for you</h3>
                                <div className="flex flex-wrap gap-2">
                                    {allTags.map(t => (
                                        <span key={t} onClick={() => { setActiveTag(t); setCurrentTab('home'); }} className="px-5 py-3 rounded-[24px] bg-[var(--bg-input)] text-[var(--text-main)] font-bold text-sm cursor-pointer hover:bg-[var(--border-color)] spring-anim active-bounce touch-ripple">
                                            {t}
                                        </span>
                                    ))}
                                    {allTags.length === 0 && <p className="text-xs text-[var(--text-muted)] p-2">Start adding tags to your items to see them here.</p>}
                                </div>
                            </div>
                        ) : (
                            <PinterestMasonry items={filteredItems} onSelect={setSelectedItem} onUpdate={handleUpdate} />
                        )}
                    </div>
                )}

                {/* === WATCH TAB (Reels / Videos) === */}
                {currentTab === 'watch' && (
                    <WatchSection items={items} onUpdate={handleUpdate} onSelectVideo={setSelectedItem} onBack={() => setCurrentTab('home')} />
                )}

                {/* === SETTINGS TAB === */}
                {currentTab === 'settings' && (
                    <div className="max-w-4xl mx-auto p-6 pt-10 animate-in space-y-10 pb-32">
                        <h1 className="text-3xl font-black text-[var(--text-main)] syne">Settings</h1>
                        
                        <div className="space-y-4">
                            <h3 className="text-xs uppercase tracking-widest font-bold text-[var(--text-muted)]">Appearance</h3>
                            <div className="flex gap-3">
                                <button onClick={()=>setTheme('light')} className={`flex-1 p-5 rounded-[24px] border active-bounce touch-ripple ${theme==='light' ? 'border-[var(--text-main)] bg-[var(--bg-input)] shadow-md' : 'border-[var(--border-color)]'} flex flex-col items-center gap-2 spring-anim`}>
                                    <Sun className="w-6 h-6"/> <span className="text-xs font-bold">Light</span>
                                </button>
                                <button onClick={()=>setTheme('dark')} className={`flex-1 p-5 rounded-[24px] border active-bounce touch-ripple ${theme==='dark' ? 'border-[var(--text-main)] bg-[var(--bg-input)] shadow-md' : 'border-[var(--border-color)]'} flex flex-col items-center gap-2 spring-anim`}>
                                    <Moon className="w-6 h-6"/> <span className="text-xs font-bold">Dark</span>
                                </button>
                            </div>

                            <div className="p-5 rounded-[24px] bg-[var(--bg-input)] border border-[var(--border-color)] shadow-inner">
                                <div className="flex gap-4 overflow-x-auto hide-scroll py-2">
                                    {Object.entries(THEMES).map(([name, rgb]) => (
                                        <button key={name} onClick={() => setAppColor(name)}
                                            className={`w-12 h-12 rounded-full shrink-0 spring-anim active-bounce touch-ripple ${appColor === name ? 'scale-110 border-4 border-[var(--bg-base)] shadow-[0_0_0_2px_var(--text-main)]' : 'scale-100'}`}
                                            style={{ background: `rgb(${rgb})` }} 
                                        />
                                    ))}
                                </div>
                            </div>
                        </div>

                        <div className="space-y-4">
                            <h3 className="text-xs uppercase tracking-widest font-bold text-[var(--text-muted)]">Default View</h3>
                            <div className="grid grid-cols-2 sm:grid-cols-4 gap-3">
                                {[{id:'grid', icon:LayoutGrid, label:'Grid'}, {id:'compact', icon:Grid3X3, label:'Compact'}, {id:'list', icon:List, label:'List'}, {id:'bento', icon:Layout, label:'Bento'}].map(l => (
                                    <button key={l.id} onClick={()=>setLayoutMode(l.id)} className={`p-4 rounded-[24px] border active-bounce touch-ripple ${layoutMode===l.id ? 'border-[var(--text-main)] bg-[var(--bg-input)] text-[var(--text-main)] shadow-md' : 'border-[var(--border-color)] text-[var(--text-muted)]'} flex flex-col items-center gap-2 spring-anim`}>
                                        <l.icon className="w-6 h-6"/> <span className="text-xs font-bold">{l.label}</span>
                                    </button>
                                ))}
                            </div>
                        </div>

                        <div className="space-y-4">
                            <h3 className="text-xs uppercase tracking-widest font-bold text-[var(--text-muted)]">Data Control</h3>
                            <button onClick={()=>{
                                const dataStr = "data:text/json;charset=utf-8," + encodeURIComponent(JSON.stringify(items, null, 2));
                                const a = document.createElement('a'); a.href = dataStr; a.download = "nexus_backup.json"; a.click();
                            }} className="w-full p-5 rounded-[24px] bg-[var(--bg-input)] border border-[var(--border-color)] text-[var(--text-main)] font-bold flex justify-center items-center gap-2 spring-anim active-bounce touch-ripple">
                                <Download className="w-5 h-5"/> Export Vault Data
                            </button>
                        </div>
                    </div>
                )}

                {/* === BOTTOM NAVIGATION === */}
                <nav className="bottom-nav">
                    <div onClick={()=>setCurrentTab('home')} className={`nav-item touch-ripple ${currentTab==='home' ? 'active' : ''}`}>
                        <HomeIcon className={currentTab==='home' ? 'fill-[var(--text-main)] w-7 h-7' : 'w-7 h-7'} strokeWidth={currentTab==='home'? 0 : 2} />
                    </div>
                    <div onClick={()=>{setCurrentTab('search'); setSearch('');}} className={`nav-item touch-ripple ${currentTab==='search' ? 'active' : ''}`}>
                        <Search className="w-7 h-7" strokeWidth={currentTab==='search'? 3 : 2} />
                    </div>
                    <div onClick={() => setShowAdd(true)} className="nav-item">
                        <div className="nav-fab shadow-xl shadow-black/20 touch-ripple"><Plus className="w-6 h-6" strokeWidth={3} /></div>
                    </div>
                    <div onClick={()=>setCurrentTab('watch')} className={`nav-item touch-ripple ${currentTab==='watch' ? 'active' : ''}`}>
                        <Clapperboard className={currentTab==='watch' ? 'fill-[var(--text-main)] w-7 h-7' : 'w-7 h-7'} strokeWidth={currentTab==='watch'? 0 : 2} />
                    </div>
                    <div onClick={()=>setCurrentTab('settings')} className={`nav-item touch-ripple ${currentTab==='settings' ? 'active' : ''}`}>
                        <User className={currentTab==='settings' ? 'fill-[var(--text-main)] w-7 h-7' : 'w-7 h-7'} strokeWidth={currentTab==='settings'? 0 : 2} />
                    </div>
                </nav>

                {/* Modal & Viewer Overlays */}
                {showAdd && <AddModal onClose={() => setShowAdd(false)} onSave={handleSave} />}
                <ItemViewer item={selectedItem} allItems={items} onClose={() => setSelectedItem(null)} onUpdate={handleUpdate} onDelete={handleDelete} onAI={handleAI} onTagClick={(t) => { setActiveTag(t); setCurrentTab('home'); }} />
            </div>
        );
    }
    createRoot(document.getElementById('root')).render(<App />);
</script>
